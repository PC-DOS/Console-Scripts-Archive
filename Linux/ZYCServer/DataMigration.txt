# Old disk mounted to: /data_bak
# New disk mounted to: /data

# Rsync copying
sudo rsync -avHAX /data_bak /data

# Remove virtual RAID
# Check info
sudo mdadm -D /dev/mdXXX
# Stop virtual RAID
sudo mdadm -S /dev/mdXXX
# New RAID
sudo mdadm --create --verbose /dev/md/sas_raid0_0 --level=0 --raid-devices=5 /dev/sd[c-g]
